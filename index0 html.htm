            { decade: "1799-1809", title: "Napoleon's Campaign", era: "ottoman", color: "#4a77b4",

              description: "First Intifada (1987-1993) and Oslo Accords (1993) establish Palestinian Authority with limited self-rule in parts of Occupied Territories." },

              description: "Second Intifada (2000-2005) results in significant casualties. Israel constructs separation barrier and expands settlements." },

              description: "Continued expansion of Israeli settlements. Gaza conflicts in 2012, 2014. Palestinian efforts for international recognition." },

        // Generate books

        const booksContainer = document.getElementById('books-container');

        const bookInfo = document.getElementById('book-info');

        const bookTitle = document.getElementById('book-title');

        const bookDescription = document.getElementById('book-description');

 

        historicalData.forEach(data => {

            const book = document.createElement('div');

            book.className = 'book';

            book.style.backgroundColor = data.color;

            book.textContent = `${data.decade}: ${data.title}`;

            book.setAttribute('data-era', data.era);

           

            book.addEventListener('click', () => {

                // Remove selected class from all books

                document.querySelectorAll('.book').forEach(b => b.classList.remove('selected'));

               

                // Add selected class to clicked book

                book.classList.add('selected');

               

                // Update book info

                bookTitle.textContent = `${data.decade}: ${data.title}`;

                bookDescription.textContent = data.description;

                bookInfo.style.display = 'block';

            });

           

            booksContainer.appendChild(book);

        });

 

        // Era filter buttons

        const eraButtons = document.querySelectorAll('.era-btn');

        eraButtons.forEach(button => {

            button.addEventListener('click', () => {

                // Remove active class from all buttons

                eraButtons.forEach(btn => btn.classList.remove('active'));

               

                // Add active class to clicked button

                button.classList.add('active');

               

                const era = button.getAttribute('data-era');

               

                // Show/hide books based on era

                document.querySelectorAll('.book').forEach(book => {

                    if (era === 'all' || book.getAttribute('data-era') === era) {

                        book.style.display = 'inline-flex';

                    } else {

                        book.style.display = 'none';

                    }

                });

            });

        });

 

        // Decade navigation

        const decadeButtons = document.querySelectorAll('.decade-btn');

        decadeButtons.forEach(button => {

            button.addEventListener('click', () => {

                const decade = button.getAttribute('data-decade');

               

                // Scroll to the approximate position of that century

                const bookshelf = document.querySelector('.bookshelf');

                const scrollPosition = (parseInt(decade) - 1100) / 10 * 80;

                bookshelf.scrollTo({

                    left: scrollPosition,

                    behavior: 'smooth'

                });

            });

        });
